import { useState, useEffect } from 'react';
import { useLawnCastStore } from '../models/store';
import {
	Table,
	TableBody,
	TableCell,
	TableContainer,
	TableHead,
	TableRow,
	Paper,
	Typography,
	IconButton,
	Dialog,
	DialogTitle,
	DialogContent,
	DialogActions,
	Button,
	TextField,
	Tooltip,
} from '@mui/material';
import EditIcon from '@mui/icons-material/Edit';
import { startOfWeek, addDays, format } from 'date-fns';
 
function getCurrentWeekDates(): string[] {
	const start = startOfWeek(new Date(), { weekStartsOn: 0 }); // Sunday
	return Array.from({ length: 7 }, (_, i) =>
		format(addDays(start, i), 'yyyy-MM-dd')
	);
}
 
export default function WeekLogTable() {
	const entries = useLawnCastStore(s => s.entries);
	const setEntry = useLawnCastStore(s => s.setEntry);
	const weekDates = getCurrentWeekDates();
 
	const [dialogOpen, setDialogOpen] = useState(false);
	const [editDate, setEditDate] = useState<string | null>(null);
	const [minutes, setMinutes] = useState('');
	const [error, setError] = useState('');
	const [isOnline, setIsOnline] = useState(navigator.onLine);
 
	useEffect(() => {
		const handleOnline = () => setIsOnline(true);
		const handleOffline = () => setIsOnline(false);
		window.addEventListener('online', handleOnline);
		window.addEventListener('offline', handleOffline);
		return () => {
			window.removeEventListener('online', handleOnline);
			window.removeEventListener('offline', handleOffline);
		};
	}, []);
 
	const handleEdit = (date: string, value: number) => {
		setEditDate(date);
		setMinutes(value.toString());
		setError('');
		setDialogOpen(true);
	};
 
	const handleClose = () => {
		setDialogOpen(false);
		setEditDate(null);
		setMinutes('');
		setError('');
	};
 
	const handleSave = () => {
		const val = parseInt(minutes, 10);
		if (isNaN(val) || val < 0 || val > 240) {
			setError('Enter a value between 0 and 240');
			return;
		}
		if (editDate) {
			setEntry(editDate, val);
		}
		handleClose();
	};
 
	return (
		<>
			<TableContainer component={Paper} variant="outlined" sx={{ mt: 2 }}>
				<Table size="small" aria-label="Week log table">
					<TableHead>
						<TableRow>
							<TableCell>
								<Typography fontWeight={700}>Day</Typography>
							</TableCell>
							<TableCell align="right">
								<Typography fontWeight={700}>
									Minutes
								</Typography>
							</TableCell>
							<TableCell align="right" />
						</TableRow>
					</TableHead>
					<TableBody>
						{weekDates.map(date => {
							const entry = entries[date];
							const dayLabel = format(new Date(date), 'EEEE');
							return (
								<TableRow key={date}>
									<TableCell>{dayLabel}</TableCell>
									<TableCell align="right">
										{entry ? entry.minutes : 0}
									</TableCell>
									<TableCell align="right">
										<Tooltip
											title={
												isOnline
													? ''
													: 'Editing is disabled while offline'
											}
										>
											<span>
												<IconButton
													aria-label={`Edit minutes for ${dayLabel}`}
													onClick={() =>
														handleEdit(
															date,
															entry
																? entry.minutes
																: 0
														)
													}
													disabled={!isOnline}
												>
													<EditIcon />
												</IconButton>
											</span>
										</Tooltip>
									</TableCell>
								</TableRow>
							);
						})}
					</TableBody>
				</Table>
			</TableContainer>
			<Dialog open={dialogOpen} onClose={handleClose}>
				<DialogTitle>Edit Minutes</DialogTitle>
				<DialogContent>
					<TextField
						autoFocus
						margin="dense"
						label="Minutes"
						type="number"
						fullWidth
						value={minutes}
						onChange={e => setMinutes(e.target.value)}
						inputProps={{ min: 0, max: 240 }}
						error={!!error}
						helperText={error}
						disabled={!isOnline}
					/>
					{!isOnline && (
						<Typography
							color="error"
							variant="body2"
							sx={{ mt: 1 }}
						>
							Editing is disabled while offline.
						</Typography>
					)}
				</DialogContent>
				<DialogActions>
					<Button onClick={handleClose}>Cancel</Button>
					<Button
						onClick={handleSave}
						variant="contained"
						disabled={!isOnline}
					>
						Save
					</Button>
				</DialogActions>
			</Dialog>
		</>
	);
}
